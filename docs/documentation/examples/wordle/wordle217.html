WITH __stage0 AS (
  SELECT 
     upper(words.word) as word
  FROM `malloy-data.malloytest.words` as words
  WHERE ((length(words.word))=5)and REGEXP_CONTAINS(words.word, r'^[a-z]{5}$')
)
, __stage1 AS (
  SELECT
    group_set,
    base.word as word__0,
    CASE WHEN group_set=1 THEN
      substr(base.word,numbers_0.num,1)
      END as letter__1,
    CASE WHEN group_set=1 THEN
      numbers_0.num
      END as position__1
  FROM __stage0 as base
  LEFT JOIN `malloy-data.malloytest.numbers` AS numbers_0
    ON 1=1 AND (numbers_0.num<=5)
  CROSS JOIN (SELECT row_number() OVER() -1  group_set FROM UNNEST(GENERATE_ARRAY(0,1,1)))
  GROUP BY 1,2,3,4
)
, __stage2 AS (
  SELECT
    word__0 as word,
    ARRAY_AGG(CASE WHEN group_set=1 THEN STRUCT(
      letter__1 as letter, 
      position__1 as position
      ) END IGNORE NULLS  ORDER BY  position__1 ASC ) as letters
  FROM __stage1
  GROUP BY 1
  ORDER BY 1 asc
)
, __stage3 AS (
  SELECT
    group_set,
    letters_0.letter as letter__0,
    letters_0.position as position__0,
    CASE WHEN group_set=0 THEN
      COUNT(DISTINCT wordle.word)
      END as word_count__0,
    CASE WHEN group_set=1 THEN
      wordle.word
      END as word__1
  FROM __stage2 as wordle
  LEFT JOIN UNNEST(wordle.letters) as letters_0
  CROSS JOIN (SELECT row_number() OVER() -1  group_set FROM UNNEST(GENERATE_ARRAY(0,1,1)))
  WHERE (COALESCE(NOT(REGEXP_CONTAINS(wordle.word, r'(S|ED)$')),FALSE))
  AND (COALESCE(NOT(REGEXP_CONTAINS(wordle.word, r'(S|ED)$')),FALSE))
  GROUP BY 1,2,3,5
)
, __stage4 AS (
  SELECT
    letter__0 as letter,
    position__0 as position,
    ANY_VALUE(CASE WHEN group_set=0 THEN word_count__0 END) as word_count,
    ARRAY_AGG(CASE WHEN group_set=1 THEN STRUCT(
      word__1 as word
      ) END IGNORE NULLS  ORDER BY  word__1 asc ) as words_list
  FROM __stage3
  GROUP BY 1,2
  ORDER BY 3 desc
)
SELECT 
   words_list_0.word as word,
   COALESCE(CAST((
    (
      SUM(DISTINCT
        (CAST(ROUND(COALESCE(base.word_count,0)*(1*1.0), 9) AS NUMERIC) +

FROM (SELECT GENERATE_UUID() as __distinct_key, * FROM __stage4 as x) as base
LEFT JOIN UNNEST(base.words_list) as words_list_0
GROUP BY 1
ORDER BY 2 desc

source: words is table('malloy-data.malloytest.words') {
  query: five_letter_words is {

    project: word is upper(word)
  }
}

// Cross join numbers
source: numbers is table('malloy-data.malloytest.numbers') {
  where: num <= 5
}

// Build a new table of word and each letter in position
query: words_and_position is from(words->five_letter_words) {
  // Cross join is missing at the moment
  join_cross: numbers
} -> {
  group_by: word
  nest: letters is {
    order_by: 2
    group_by:
      letter is substr(word, numbers.num, 1)
      position is numbers.num
  }
}

// Build a word finder that can generate a score best available guess
source: wordle is from(-> words_and_position) {
  where: word !~ r'(S|ED)$'
  measure: word_count is count()

  query: find_words is {
    group_by:
      letters.letter
      letters.position
    aggregate: word_count
    nest: words_list is {
      group_by: word
    }
  } -> {
    group_by: words_list.word
    aggregate: score is word_count.sum()
  }
